  <script>

    // ========== Utilities ==========

    const $ = (id) => document.getElementById(id);

    const FIELDS = ['dt','year','teacher','subject','classSet','room','period','behaviour','engagement','notes','strengths','areas','actions','email','tick_lo','tick_retrieval','tick_questioning','tick_afl','tick_send','tick_literacy','tick_homework','tick_routines'];

    let CURRENT_ID = null; // null => new entry

 

    function toLocalISOString(dt){

      const pad=(n)=>String(n).padStart(2,'0');

      const y=dt.getFullYear();

      const m=pad(dt.getMonth()+1);

      const d=pad(dt.getDate());

      const h=pad(dt.getHours());

      const mi=pad(dt.getMinutes());

      return `${y}-${m}-${d}T${h}:${mi}`;

    }

 

    function getEntries(){

      try{ return JSON.parse(localStorage.getItem('lw_entries_v1')||'[]'); }catch{return []}

    }

    function setEntries(arr){ localStorage.setItem('lw_entries_v1', JSON.stringify(arr)); }

 

    function loadFormFromLocal(){

      const raw = localStorage.getItem('lw_form_v1');

      if(!raw){ $('dt').value = toLocalISOString(new Date()); return; }

      try{

        const f = JSON.parse(raw);

        for(const k of FIELDS){

          const el = $(k); if(!el) continue;

          if(el.type==='checkbox'){ el.checked = !!f[k]; }

          else if(f[k]!==undefined) el.value = f[k];

        }

      }catch{}

    }

    function saveFormSnapshot(){

      const f = {};

      for(const k of FIELDS){ const el=$(k); if(!el) continue; f[k]= (el.type==='checkbox') ? el.checked : el.value; }

      localStorage.setItem('lw_form_v1', JSON.stringify(f));

      return f;

    }

 

    function clearForm(){

      CURRENT_ID = null;

      ['teacher','subject','classSet','room','period','behaviour','engagement','notes','strengths','areas','actions'].forEach(k=>$(k).value='');

      ['tick_lo','tick_retrieval','tick_questioning','tick_afl','tick_send','tick_literacy','tick_homework','tick_routines'].forEach(k=>$(k).checked=false);

      $('year').value='Y11';

      $('dt').value=toLocalISOString(new Date());

      saveFormSnapshot();

    }

 

    function entryFromForm(){

      const f = saveFormSnapshot();

      return {

        id: CURRENT_ID || ('e'+Date.now()),

        ...f

      };

    }

 

    function fillForm(e){

      CURRENT_ID = e.id;

      for(const k of FIELDS){ const el=$(k); if(!el) continue; if(el.type==='checkbox') el.checked=!!e[k]; else if(e[k]!==undefined) el.value=e[k]; }

      saveFormSnapshot();

      updateMailto();

      window.scrollTo({top:0,behavior:'smooth'});

    }

 

    function entryToText(e){

      const dtNice = e.dt ? new Date(e.dt).toLocaleString(undefined,{dateStyle:'medium',timeStyle:'short'}) : '';

      const ticks = ['tick_lo','tick_retrieval','tick_questioning','tick_afl','tick_send','tick_literacy','tick_homework','tick_routines']

        .filter(k=>e[k]).map(k=>({tick_lo:'LO visible',tick_retrieval:'Retrieval starter',tick_questioning:'Effective questioning',tick_afl:'AFL checks',tick_send:'SEND scaffolds',tick_literacy:'Literacy focus',tick_homework:'Homework set',tick_routines:'Strong routines'})[k]).join(', ');

      return [

        'Learning Walk Note',

        `Date/Time: ${dtNice}`,

        `Year Group: ${e.year||''}`,

        `Teacher: ${e.teacher||''}`,

        `Subject: ${e.subject||''}`,

        `Class/Set: ${e.classSet||''}`,

        `Room: ${e.room||''}`,

        `Lesson/Period: ${e.period||''}`,

        `Behaviour: ${e.behaviour||''} /5`,

        `Engagement: ${e.engagement||''} /5`,

        ticks?`Quick ticks: ${ticks}`:null,

        '',

        'Observation Notes:',

        e.notes||'-',

        '',

        'Strengths:',

        e.strengths||'-',

        '',

        'Areas for Development:',

        e.areas||'-',

        '',

        'Actions / Next Steps:',

        e.actions||'-'

      ].filter(Boolean).join('

');

    }

 

    function buildMailtoFor(e){

      const subject = `Learning Walk – ${e.year||''} – ${e.teacher||'Teacher'} – ${e.subject||''} – ${e.dt?new Date(e.dt).toLocaleDateString():''}`;

      const body = entryToText(e);

      const to = e.email || '';

      return `mailto:${encodeURIComponent(to)}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;

    }

 

    function updateMailto(){

      const e = entryFromForm();

      $('mailtoBtn').setAttribute('href', buildMailtoFor(e));

    }

 

    async function copySummary(){

      const text = entryToText(entryFromForm());

      try{ await navigator.clipboard.writeText(text); alert('Summary copied to clipboard.'); }

      catch(e){ window.prompt('Copy summary:', text); }

    }

 

    function saveEntry(){

      const e = entryFromForm();

      const arr = getEntries();

      const idx = arr.findIndex(x=>x.id===e.id);

      if(idx>=0) arr[idx]=e; else arr.unshift(e);

      setEntries(arr);

      renderList();

      alert(idx>=0? 'Entry updated.' : 'Entry saved.');

    }

 

    function deleteEntry(id){

      if(!confirm('Delete this entry?')) return;

      const arr = getEntries().filter(x=>x.id!==id);

      setEntries(arr);

      if(CURRENT_ID===id) clearForm();

      renderList();

    }

 

    function renderList(){

      const q = $('search').value.toLowerCase().trim();

      const arr = getEntries().filter(e=>{

        if(!q) return true;

        const hay = [e.teacher,e.subject,e.year,e.room,e.classSet,e.notes,e.strengths,e.areas,e.actions].join(' ').toLowerCase();

        return hay.includes(q);

      });

      const tbody = $('tbody');

      tbody.innerHTML = '';

      for(const e of arr){

        const tr = document.createElement('tr');

        const dtShort = e.dt? new Date(e.dt).toLocaleString(undefined,{dateStyle:'medium', timeStyle:'short'}):'';

        const summary = (e.notes||'').slice(0,80) + ((e.notes||'').length>80?'…':'');

        tr.innerHTML = `

          <td>${dtShort}</td>

          <td>${e.year||''}</td>

          <td>${e.teacher||''}</td>

          <td>${e.subject||''}</td>

          <td>${e.room||''}</td>

          <td>${summary}</td>

          <td>

            <button class="btn secondary" data-act="view" data-id="${e.id}">View/Edit</button>

            <a class="btn" href="${buildMailtoFor(e)}">Email</a>

            <button class="btn warn" data-act="copy" data-id="${e.id}">Copy</button>

            <button class="btn danger" data-act="del" data-id="${e.id}">Delete</button>

          </td>`;

        tbody.appendChild(tr);

      }

    }

 

    function exportCsv(){

      const arr = getEntries();

      const headers = ['DateTime','Year','Teacher','Subject','ClassSet','Room','Period','Behaviour','Engagement','LO','Retrieval','Questioning','AFL','SEND','Literacy','Homework','Routines','Notes','Strengths','Areas','Actions'];

      const rows = arr.map(e=>[

        e.dt||'',e.year||'',e.teacher||'',e.subject||'',e.classSet||'',e.room||'',e.period||'',e.behaviour||'',e.engagement||'',

        e.tick_lo?1:0,e.tick_retrieval?1:0,e.tick_questioning?1:0,e.tick_afl?1:0,e.tick_send?1:0,e.tick_literacy?1:0,e.tick_homework?1:0,e.tick_routines?1:0,

        (e.notes||'').replace(/

/g,' '),(e.strengths||'').replace(/

/g,' '),(e.areas||'').replace(/

/g,' '),(e.actions||'').replace(/

/g,' ')

      ]);

      const csv = [headers.join(','), ...rows.map(r=>r.map(v=>`"${String(v).replace(/"/g,'""')}"`).join(','))].join('

');

      const blob = new Blob([csv],{type:'text/csv'});

      const url = URL.createObjectURL(blob);

      const a = document.createElement('a');

      a.href=url; a.download='learning-walks.csv'; a.click();

      URL.revokeObjectURL(url);

    }

 

    // ===== Event bindings =====

    FIELDS.forEach(id=>{ const el=$(id); if(!el) return; el.addEventListener('input', ()=>{ saveFormSnapshot(); updateMailto(); }); });

    $('copyBtn').addEventListener('click', copySummary);

    $('clearBtn').addEventListener('click', ()=>{ if(confirm('Clear the form?')){ clearForm(); updateMailto(); }});

    $('mailtoBtn').addEventListener('click', ()=>{ updateMailto(); });

    $('saveEntryBtn').addEventListener('click', saveEntry);

    $('newEntryBtn').addEventListener('click', ()=>{ clearForm(); alert('New entry started.'); });

    $('search').addEventListener('input', renderList);

    $('exportCsvBtn').addEventListener('click', exportCsv);

 

    $('tbody').addEventListener('click', (e)=>{

      const btn = e.target.closest('button'); if(!btn) return;

      const id = btn.getAttribute('data-id');

      const act = btn.getAttribute('data-act');

      const list = getEntries();

      const entry = list.find(x=>x.id===id);

      if(!entry) return;

      if(act==='view') fillForm(entry);

      else if(act==='del') deleteEntry(id);

      else if(act==='copy') { const text=entryToText(entry); navigator.clipboard.writeText(text).then(()=>alert('Copied.')); }

    });

 

    // ===== Init =====

    loadFormFromLocal();

    if(!$('dt').value) $('dt').value = toLocalISOString(new Date());

    updateMailto();

    renderList();

  </script>
